```model ComplexPoles

parameter Real omegan "Natural frequency, 0.1";
parameter Real mu "Gain, 1";
parameter Real y_start "Output initial value, 0";
parameter Real dy_start "Slope initial value, 0";
protected Real u_pre1 "Previous input";
protected Real y_pre1 "Previous output";
equation
pre(y) = y_start;
pre(y_pre1) = dy_start;
assert(xi >= 0 and xi <= 1, "Damping coefficient xi must be between 0 and 1");
y = IndustrialControlSystems.LinearSystems.Discrete.Functions.fCmplx((alfa,u,pre(u),pre(u_pre1),pre(y),pre(y_pre1),Ts,mu,xi,omegan));
u_pre1 = u;
y_pre1 = y;
sample(0, Ts);
end ComplexPoles;
```
