```modelica
model AmbientCondition
  // Parameter representing the ambient temperature
  parameter Modelica.SIunits.Temperature T_amb = 293.15 "Ambient temperature (default: 20°C)";

  // Heat transfer interface
  Modelica.Thermal.HeatTransfer.Interfaces.HeatPort_a node "Heat transfer interface";

equation
  // Ensure the temperature at the heat port is equal to the ambient temperature
  node.T = T_amb;
  // No heat flow into or out of the infinite reservoir
  node.Q_flow = 0;
end AmbientCondition;
```
