```modelica
model Add
  extends IndustrialControlSystems.MathOperations.IntegerType.Interfaces.IntSimpleOperation;

protected 
  Integer u1, u2;

equation
  // Assign inputs to u1 and u2
  u1 = u[1];

    end when;
  else
    y = u1 + u2;
  end if;

end Add;
```
