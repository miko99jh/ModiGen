```modelica
model SmoothStep
  extends IndustrialControlSystems.MathOperations.RealType.Interfaces.RealNinOperation(
    final useInputs = false, 
    final FixedPoint = false, 
    final Nbit = 1, 
    final scaleFactor = 1, 
    final MAX = 1, 
    final MIN = 0);

  // Parameters
  parameter Real tin = 10 "Initial time for the smooth step [s]";
  parameter Real alfa(min = 0, max = 1) = 0.1 "Portion of time for parabolic junctions [dimensionless, range: 0 to 1]";
  parameter Real yin = 0 "Output initial value [dimensionless]";
  parameter Real yfin = 1 "Output final value [dimensionless]";
  parameter Real m = 0.1 "Slope of the ramp [dimensionless]";

  // Variables
  Real y "Output of the smooth step generator";

protected 
  Real Ufp[nInput] = zeros(nInput) "Fixed-point inputs (not used)";
  Real Yfp = 0 "Fixed-point output (not used)";

equation 
  // Ensure alfa is within the valid range
  assert(alfa >= 0 and alfa <= 1, "alfa out of range [0,1]");

  // Smooth step function
  y = IndustrialControlSystems.MathOperations.RealType.Functions.fSmoothStep(
    time, tin, alfa, m, yin, yfin);
  
  // Fixed-point inputs and outputs are not used
  Ufp = zeros(nInput);
  Yfp = 0;

end SmoothStep;
```
